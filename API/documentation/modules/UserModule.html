import { Module } from '@nestjs/common';
import { UserService } from './service/user.service';
import { UserController } from './controller/user.controller';
import { TypeOrmModule } from '@nestjs/typeorm';
import { UserEntity } from './models/user.entity';
import { AuthModule } from 'src/auth/auth.module';
import { LogModule } from 'src/log/log.module';
import { LogService } from 'src/log/service/log.service';
import { LogEntity } from 'src/log/models/log.entity';
@Module({
    imports: [
        TypeOrmModule.forFeature([UserEntity,LogEntity]),
        AuthModule,
        LogModule
    ],
    providers: [UserService, LogService],
    controllers: [UserController],
    exports: [UserService, LogService]
})
export class UserModule { }
