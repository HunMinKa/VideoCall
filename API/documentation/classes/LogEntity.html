import { Entity, PrimaryGeneratedColumn, Column, BeforeInsert, OneToMany, PrimaryColumn, CreateDateColumn, UpdateDateColumn} from "typeorm";


@Entity('api_log_history')
export class LogEntity {

    @PrimaryGeneratedColumn()
    seq: number;

    @CreateDateColumn({ name: 'request_date' , default: 'NOW()' })
    request_date: Date;

    @Column()
    request_header: string;

    @Column()
    request_key: string;

    @Column()
    request_body: string;

    @UpdateDateColumn()
    response_date: Date;

    @Column()
    response_header: string;

    @Column()
    response_key: string;

    @Column()
    response_body: string;

    @CreateDateColumn({ name: 'input_utc_dt' })
    input_utc_dt: Date;

    @PrimaryGeneratedColumn()
    measure_seq: number;

    @Column()
    measure_table: string;
}
