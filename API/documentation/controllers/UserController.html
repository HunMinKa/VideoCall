import { Controller, Post, Body, Get, Param, Delete, Put, UseGuards, Query, UseInterceptors, UploadedFile, HttpStatus , Res , Req} from '@nestjs/common';
import { UserService } from '../service/user.service';
import { UserI } from '../models/user.interface';
import { Observable, of, forkJoin  } from 'rxjs';
import { catchError, map, tap } from 'rxjs/operators';
import { promises } from 'dns';
import CryptoJs from 'crypto-js'
import { JwtAuthGuard } from 'src/auth/guards/jwt-guard';
import { RefreshToken } from 'src/auth/guards/refresh-token';
import { UserIsUserGuard } from 'src/auth/guards/UserIsUser.guard';
import { JwtService } from '@nestjs/jwt';
import { ConfigModule, ConfigService } from '@nestjs/config';
import { config } from 'process';
import { Response, Request } from 'express';
import { LogService } from 'src/log/service/log.service';


import * as CryptoJS from 'crypto-js'
import { AES, enc } from 'crypto-js'

import { ApiTags, ApiOkResponse, ApiForbiddenResponse,ApiCreatedResponse } from '@nestjs/swagger'


@ApiTags()
@Controller('users')
export class UserController {
    constructor(private userService: UserService
                , private logService: LogService
                , private configService: ConfigService) {
      
    }
    public currentDate = new Date();

    @Post()
    @ApiCreatedResponse({ description: 'Success' })
    @ApiForbiddenResponse({ description: 'Forbidden' })
    create(@Body() user: UserI): Observable<UserI | Object> {
        return this.userService.create(user).pipe(
            map((user: UserI) => user),
            catchError(err => of({ error: err.message }))
        );
    }

    @UseGuards(JwtAuthGuard,RefreshToken)
    @Get()
    @ApiOkResponse({ description: 'Success' })
    @ApiForbiddenResponse({ description: 'Forbidden' })
    findAll(@Req() req: Request, @Res() res: Response): Observable<any> {
        console.log(this.userService.findAll(req))
        return this.userService.findAll(req).pipe(
            map((users: UserI[]) => {
           
                users.forEach(function (v) { delete v.password });

                this.logService.getData(req, res, req?.body, {
                    request_date: this.currentDate,
                    value: this.encryptData(users, req)
                })

                return res.status(HttpStatus.OK).json(this.encryptData(users, req))
            })
        );
    }

    @UseGuards(JwtAuthGuard,RefreshToken)
    @Get(':id')
    @ApiOkResponse({ description: 'Success' })
    @ApiForbiddenResponse({ description: 'Forbidden' })
    findOne(@Param() params): Observable<UserI> {
        
        return this.userService.findOne(params.id).pipe(
            map((user: UserI) => {
                const { password, ...result } = user;
                return result;
            })
        )
    }
 
    @Post('login')
    @ApiCreatedResponse({ description: 'Success' })
    @ApiForbiddenResponse({ description: 'Forbidden' })
    login(@Body() user: UserI, @Req() req: Request, @Res() res: Response): Observable<Object> {
        return this.userService.login(user).pipe(
            map((jwt: string) => {

                this.logService.create(req, res, req?.body,{
                    request_date: this.currentDate,
                    value: jwt
                })

                return res.status(HttpStatus.OK).json({
                     access_token: jwt
                });

            })
        )
    }

    @Delete(':id')
    @ApiOkResponse({ description: 'Success' })
    @ApiForbiddenResponse({ description: 'Forbidden' })
    deleteOne(@Param('id') id: string): Observable<any> {
        return this.userService.deleteOne(Number(id));
    }

    @Put(':id')
    @ApiOkResponse({ description: 'Success' })
    @ApiForbiddenResponse({ description: 'Forbidden' })
    updateOne(@Param('id') id: string, @Body() user: UserI): Observable<any> {
        return this.userService.updateOne(Number(id), user);
    }


    encryptData(data: any, @Req() req: Request): string {

        // Encrypt
        var ciphertext = CryptoJS.AES.encrypt(JSON.stringify(data), req?.rawHeaders[1].split(" ")[1]).toString();

        //Decrypt
        var bytes = CryptoJS.AES.decrypt(ciphertext, req?.rawHeaders[1].split(" ")[1]);
        var decryptedData = JSON.parse(bytes.toString(CryptoJS.enc.Utf8));


        return ciphertext;
    }
 
}
